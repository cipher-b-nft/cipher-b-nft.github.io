import numpy as np 
import pandas as pd 
import pydub 
import PIL 
from PIL import Image 
import warnings

def decode(x): 
  
  a = []

  with warnings.catch_warnings():
    warnings.filterwarnings('error')
    try:
      for i in x:
        n = i[0]*i[1]+i[2]
        if i[0]>i[1]: n = n*-1
        a = n
    except Warning as e:
      n = int(x[0][0])*int(x[0][1])+int(x[0][2])
      if int(x[0][0])>int(x[0][1]): n = n*-1
      a = n    

  return a
  
img = PIL.Image.open('nft.png') 
a1 = np.asarray(img) 
a2 = a1.reshape(1,-1) 
a3 = a2.reshape(-1,3)

arr3 = [] 
sr = 44100
for ln in a3: 
	arr3.append(decode([ln]))
	arr3 = np.array(arr3) 

out = np.array(arr3).reshape(-1,2)
	
channels = 2 if (out.ndim == 2 and out.shape[1] == 2) else 1 
y = np.int16(out) 
song = pydub.AudioSegment(y.tobytes(), frame_rate=sr, sample_width=2, channels=channels) 
song.export('out.mp3', format="mp3", bitrate="320k") 
print('Finished!')
